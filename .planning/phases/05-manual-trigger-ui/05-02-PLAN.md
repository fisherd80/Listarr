```python
_running_jobs = {}
_jobs_lock = threading.Lock()

# Lazy-initialized executor (created on first use)
_executor = None

def get_executor():
    """Get or create the ThreadPoolExecutor."""
    global _executor
    if _executor is None:
        _executor = ThreadPoolExecutor(max_workers=3, thread_name_prefix='listarr_import_')
    return _executor
```

```python
    """Background job that runs the actual import."""
    with app.app_context():

            result = import_list(list_id)

                _running_jobs[list_id] = {

                    'status': 'error',
```

```python
@bp.route("/lists/<int:list_id>/run", methods=["POST"])
@csrf.exempt
def run_list_import(list_id):
    """
    Manually trigger import for a list (async).
    Returns 202 Accepted immediately, job runs in background.

    Returns:
        202: {"success": true, "job_id": list_id, "status": "started"}
        400: {"success": false, "error": "..."} if already running or inactive
        404: {"success": false, "error": "..."} if not found
    """
    # Fetch list by ID
    list_obj = List.query.get(list_id)
    if not list_obj:
        return jsonify({
            "success": False,
            "error": f"List with ID {list_id} not found"
        }), 404

    # Check if list is active
    if not list_obj.is_active:
        return jsonify({
            "success": False,
            "error": f"List '{list_obj.name}' is not active"
        }), 400

    # Check if already running
    with _jobs_lock:
        if list_id in _running_jobs and _running_jobs[list_id].get('status') == 'running':
            return jsonify({
                "success": False,
                "error": f"List '{list_obj.name}' is already running"
            }), 400

        # Mark as running
        _running_jobs[list_id] = {
            'status': 'running',
            'started_at': datetime.now(timezone.utc),
            'result': None
        }

    # Submit to executor (pass app for context)
    get_executor().submit(_run_import_job, list_id, current_app._get_current_object())

    return jsonify({
        "success": True,
        "job_id": list_id,
        "status": "started"
    }), 202
```

```python
@bp.route("/lists/<int:list_id>/status", methods=["GET"])
def get_list_status(list_id):
    """
    Get the status of a list import job for polling.

    Returns JSON with:
    - list_id: int
    - status: 'idle'|'running'|'completed'|'error'
    - last_run_at: ISO timestamp string or null
    - result: ImportResult dict (if completed)
    - error: string (if error)
    """
    list_obj = List.query.get(list_id)
    if not list_obj:
        return jsonify({
            "error": f"List with ID {list_id} not found"
        }), 404

    # Check in-memory job state
    with _jobs_lock:
        job = _running_jobs.get(list_id)

    if job and job.get('status') == 'running':
        return jsonify({
            "list_id": list_id,
            "status": "running",
            "last_run_at": list_obj.last_run_at.isoformat() if list_obj.last_run_at else None
        })
    elif job and job.get('status') == 'completed':
        # Clear job from memory after reporting (client will stop polling)
        result = job.get('result')
        with _jobs_lock:
            del _running_jobs[list_id]
        return jsonify({
            "list_id": list_id,
            "status": "completed",
            "last_run_at": list_obj.last_run_at.isoformat() if list_obj.last_run_at else None,
            "result": result
        })
    elif job and job.get('status') == 'error':
        error = job.get('error')
        with _jobs_lock:
            del _running_jobs[list_id]
        return jsonify({
            "list_id": list_id,
            "status": "error",
            "last_run_at": list_obj.last_run_at.isoformat() if list_obj.last_run_at else None,
            "error": error
        })
    else:
        # No job running
        return jsonify({
            "list_id": list_id,
            "status": "idle",
            "last_run_at": list_obj.last_run_at.isoformat() if list_obj.last_run_at else None
        })
```

The endpoint:
- Returns 'running' if job is currently in progress
- Returns 'completed' with result when done (and clears from memory)
- Returns 'error' with error message on failure
- Returns 'idle' if no job running
- No CSRF needed (GET request)

```javascript
function getRunningJobs() {
  return JSON.parse(localStorage.getItem(RUNNING_JOBS_KEY) || '{}');
}

function setRunningJobs(jobs) {
  localStorage.setItem(RUNNING_JOBS_KEY, JSON.stringify(jobs));
}

function trackRunningJob(listId, startTime) {
  const jobs = getRunningJobs();
  jobs[listId] = {
    startedAt: startTime,
    status: 'running'
  };
  setRunningJobs(jobs);
}

function clearRunningJob(listId) {
  const jobs = getRunningJobs();
  delete jobs[listId];
  setRunningJobs(jobs);
}

function isJobRunning(listId) {
  const jobs = getRunningJobs();
  return !!jobs[listId];
}
```

3. Add polling infrastructure (track active polls to prevent duplicates):
```javascript
// Active polling timers and abort controllers
const activePolls = {};
const abortControllers = {};

function pollJobStatus(listId) {
  // Prevent duplicate polling
  if (activePolls[listId]) {
    return;
  }

  const jobs = getRunningJobs();
  const job = jobs[listId];
  if (!job) {
    return;
  }

  const startedAt = job.startedAt;
  const controller = new AbortController();
  abortControllers[listId] = controller;

  const poll = () => {
    // Check timeout
    if (Date.now() - startedAt > MAX_POLL_DURATION_MS) {
      showToast("Import taking longer than expected. Check back later.", "warning", 5000);
      clearRunningJob(listId);
      restoreButton(listId);
      delete activePolls[listId];
      delete abortControllers[listId];
      return;
    }

    fetch(`/lists/${listId}/status`, {
      signal: controller.signal
    })
      .then(res => {
        if (!res.ok) {
          throw new Error(`HTTP error! status: ${res.status}`);
        }
        return res.json();
      })
      .then(data => {
        // Status endpoint returns: idle, running, completed, error
        if (data.status === 'completed') {
          handleJobComplete(listId, data.result);
          clearRunningJob(listId);
          delete activePolls[listId];
          delete abortControllers[listId];
        } else if (data.status === 'error') {
          handleJobError(listId, data.error);
          clearRunningJob(listId);
          delete activePolls[listId];
          delete abortControllers[listId];
        } else if (data.status === 'idle') {
          // Job finished but we missed it (server restarted?) - treat as completed
          handleJobComplete(listId, null);
          clearRunningJob(listId);
          delete activePolls[listId];
          delete abortControllers[listId];
        } else {
          // Still running - continue polling
          activePolls[listId] = setTimeout(poll, POLL_INTERVAL_MS);
        }
      })
      .catch(error => {
        if (error.name === 'AbortError') {
          return; // Normal cleanup
        }
        console.error('Polling error:', error);
        // Continue polling on network errors (transient)
        activePolls[listId] = setTimeout(poll, POLL_INTERVAL_MS);
      });
  };

  // Start first poll after delay (give server time to process)
  activePolls[listId] = setTimeout(poll, POLL_INTERVAL_MS);
}

function restoreButton(listId) {
  const button = document.querySelector(`[data-run-list="${listId}"]`);
  if (button) {
    button.disabled = false;
    button.textContent = "Run";
  }
}

function handleJobComplete(listId, result) {
  restoreButton(listId);

  if (result) {
    // Show color-coded toast with result counts
    const { added_count, skipped_count, failed_count } = result;
    const message = `${added_count} added, ${skipped_count} skipped, ${failed_count} failed`;

    let type = 'success';
    if (failed_count > 0) {
      type = 'error';
    } else if (skipped_count > 0) {
      type = 'warning';
    }
    showToast(message, type, 5000);
  } else {
    // No result available (e.g., navigated away or server restarted)
    showToast("Import completed.", "success", 5000);
  }
}

function handleJobError(listId, error) {
  restoreButton(listId);
  showToast(`Import failed: ${error || 'Unknown error'}`, "error", 5000);
}
```
