@bp.route("/lists/<int:list_id>/status", methods=["GET"])
def get_list_status(list_id):
    """
    Get the status of a list for polling.

    Returns JSON with:
    - list_id: int
    - last_run_at: ISO timestamp string or null
    - is_active: bool
    """
    list_obj = List.query.get(list_id)
    if not list_obj:
        return jsonify({
            "error": f"List with ID {list_id} not found"
        }), 404

    return jsonify({
        "list_id": list_id,
        "last_run_at": list_obj.last_run_at.isoformat() if list_obj.last_run_at else None,
        "is_active": list_obj.is_active
    })
```

The endpoint:
- Returns 404 if list doesn't exist
- Returns `last_run_at` as ISO string (or null if never run)
- Returns `is_active` for client-side validation
- No CSRF needed (GET request)
- No authentication needed (homelab single-user)

function getRunningJobs() {
  return JSON.parse(localStorage.getItem(RUNNING_JOBS_KEY) || '{}');
}

function setRunningJobs(jobs) {
  localStorage.setItem(RUNNING_JOBS_KEY, JSON.stringify(jobs));
}

function trackRunningJob(listId, startTime) {
  const jobs = getRunningJobs();
  jobs[listId] = {
    startedAt: startTime,
    status: 'running'
  };
  setRunningJobs(jobs);
}

function clearRunningJob(listId) {
  const jobs = getRunningJobs();
  delete jobs[listId];
  setRunningJobs(jobs);
}

function isJobRunning(listId) {
  const jobs = getRunningJobs();
  return !!jobs[listId];
}
```

3. Add polling infrastructure (track active polls to prevent duplicates):
```javascript
// Active polling timers and abort controllers
const activePolls = {};
const abortControllers = {};

function pollJobStatus(listId) {
  // Prevent duplicate polling
  if (activePolls[listId]) {
    return;
  }

  const jobs = getRunningJobs();
  const job = jobs[listId];
  if (!job) {
    return;
  }

  const startedAt = job.startedAt;
  const controller = new AbortController();
  abortControllers[listId] = controller;

  const poll = () => {
    // Check timeout
    if (Date.now() - startedAt > MAX_POLL_DURATION_MS) {
      showToast("Import taking longer than expected. Check back later.", "warning", 5000);
      clearRunningJob(listId);
      delete activePolls[listId];
      delete abortControllers[listId];
      // Leave button disabled - user needs to refresh or check later
      return;
    }

    fetch(`/lists/${listId}/status`, {
      signal: controller.signal
    })
      .then(res => {
        if (!res.ok) {
          throw new Error(`HTTP error! status: ${res.status}`);
        }
        return res.json();
      })
      .then(data => {
        const lastRunAt = data.last_run_at ? new Date(data.last_run_at).getTime() : 0;

        if (lastRunAt >= startedAt) {
          // Job completed - fetch the full result
          handleJobComplete(listId);
          clearRunningJob(listId);
          delete activePolls[listId];
          delete abortControllers[listId];
        } else {
          // Still running - continue polling
          activePolls[listId] = setTimeout(poll, POLL_INTERVAL_MS);
        }
      })
      .catch(error => {
        if (error.name === 'AbortError') {
          return; // Normal cleanup
        }
        console.error('Polling error:', error);
        // Continue polling on network errors (transient)
        activePolls[listId] = setTimeout(poll, POLL_INTERVAL_MS);
      });
  };

  // Start first poll after delay (give server time to process)
  activePolls[listId] = setTimeout(poll, POLL_INTERVAL_MS);
}

function handleJobComplete(listId) {
  const button = document.querySelector(`[data-run-list="${listId}"]`);
  if (button) {
    button.disabled = false;
    button.textContent = "Run";
  }

  // Note: We can't get the actual ImportResult here since we navigated away
  // Just show a generic success message
  showToast("Import completed. Check results in your media server.", "success", 5000);
}
```
