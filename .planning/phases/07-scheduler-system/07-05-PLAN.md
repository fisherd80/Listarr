Add an API endpoint to return upcoming scheduled jobs.

**GET /api/dashboard/upcoming**

Returns JSON array of next 5 scheduled jobs, sorted by next run time:
```json
{
  "upcoming": [
    {
      "list_id": 1,
      "list_name": "Trending Movies",
      "service": "radarr",
      "next_run": "2026-02-05T14:00:00Z",
      "next_run_relative": "in 2 hours"
    }
  ],
  "scheduler_paused": false
}
```

Implementation:
```python
@bp.route('/api/dashboard/upcoming')
def get_upcoming_jobs():
    from listarr.models.lists_model import List
    from listarr.services.scheduler import get_next_run_time, is_scheduler_paused

    # Get all active lists with schedules
    lists = List.query.filter(
        List.schedule_cron.isnot(None),
        List.is_active == True
    ).all()

    upcoming = []
    for list_obj in lists:
        next_run = get_next_run_time(list_obj.id)
        if next_run:
            upcoming.append({
                'list_id': list_obj.id,
                'list_name': list_obj.name,
                'service': list_obj.target_service,
                'next_run': next_run.isoformat() if next_run else None,
                'next_run_relative': format_relative_time(next_run)
            })

    # Sort by next_run and take top 5
    upcoming.sort(key=lambda x: x['next_run'] or '')
    upcoming = upcoming[:5]

    return jsonify({
        'upcoming': upcoming,
        'scheduler_paused': is_scheduler_paused()
    })
```

```javascript
async function loadUpcoming() {
  try {
    const response = await fetch('/api/dashboard/upcoming');
    const data = await response.json();

    const loadingEl = document.getElementById('upcoming-loading');
    const emptyEl = document.getElementById('upcoming-empty');
    const listEl = document.getElementById('upcoming-list');
    const pausedBadge = document.getElementById('scheduler-status-badge');

    // Hide loading
    loadingEl.classList.add('hidden');

    // Show/hide paused badge
    if (data.scheduler_paused) {
      pausedBadge.classList.remove('hidden');
    } else {
      pausedBadge.classList.add('hidden');
    }

    // Show upcoming or empty state
    if (data.upcoming.length === 0) {
      emptyEl.classList.remove('hidden');
      listEl.classList.add('hidden');
    } else {
      emptyEl.classList.add('hidden');
      listEl.classList.remove('hidden');
      listEl.innerHTML = data.upcoming.map(job => `
        <li class="flex items-center justify-between">
          <div>
            <span class="text-sm font-medium text-gray-900 dark:text-gray-100">
              ${escapeHtml(job.list_name)}
            </span>
            <span class="ml-2 text-xs px-1.5 py-0.5 rounded ${job.service === 'radarr' ? 'bg-amber-100 text-amber-800 dark:bg-amber-900 dark:text-amber-200' : 'bg-blue-100 text-blue-800 dark:bg-blue-900 dark:text-blue-200'}">
              ${job.service}
            </span>
          </div>
          <span class="text-sm text-gray-500 dark:text-gray-400">
            ${escapeHtml(job.next_run_relative)}
          </span>
        </li>
      `).join('');
    }
  } catch (error) {
    console.error('Failed to load upcoming jobs:', error);
  }
}
```
