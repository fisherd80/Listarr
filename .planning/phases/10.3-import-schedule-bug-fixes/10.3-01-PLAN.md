     def get_exclusions(base_url: str, api_key: str) -> set[int]:
         """
         Fetch excluded TMDB IDs from Radarr's exclusion list.

         Args:
             base_url: Base URL of Radarr.
             api_key: Radarr API key.

         Returns:
             Set of excluded TMDB IDs, empty set on error.
