**Backend - Create preview endpoint:**

Add route `/lists/wizard/preview` (POST, returns JSON):
```python
@bp.route("/lists/wizard/preview", methods=["POST"])
def wizard_preview():
    data = request.get_json()
    service = data.get("service")  # radarr or sonarr
    preset = data.get("preset")    # preset name or None
    filters = data.get("filters", {})  # genre_ids, year_min, year_max, rating_min

    # Get TMDB API key from settings
    from listarr.models import ServiceConfig
    tmdb_config = ServiceConfig.get_by_service("TMDB")
    if not tmdb_config or not tmdb_config.api_key:
        return jsonify({"error": "TMDB not configured", "items": []})

    api_key = decrypt_data(tmdb_config.api_key)

    # Fetch based on preset or filters
    if preset in ["trending_movies", "trending_tv"]:
        items = get_trending_movies/tv(api_key)
    elif preset in ["popular_movies", "popular_tv"]:
        items = get_popular_movies/tv(api_key)
    else:
        # Build TMDB filter dict
        tmdb_filters = {}
        if filters.get("genre_ids"):
            tmdb_filters["with_genres"] = ",".join(map(str, filters["genre_ids"]))
        if filters.get("year_min"):
            tmdb_filters["primary_release_date.gte" if service=="radarr" else "first_air_date.gte"] = f"{filters['year_min']}-01-01"
        if filters.get("year_max"):
            tmdb_filters["primary_release_date.lte" if service=="radarr" else "first_air_date.lte"] = f"{filters['year_max']}-12-31"
        if filters.get("rating_min"):
            tmdb_filters["vote_average.gte"] = filters["rating_min"]

        if service == "radarr":
            items = discover_movies(api_key, tmdb_filters)
        else:
            items = discover_tv(api_key, tmdb_filters)

    # Return first 5 items for preview
    preview_items = []
    for item in items[:5]:
        preview_items.append({
            "id": item.id,
            "title": item.title if hasattr(item, 'title') else item.name,
            "year": (item.release_date or item.first_air_date or "")[:4],
            "rating": round(item.vote_average, 1) if item.vote_average else None,
        })

    return jsonify({"items": preview_items})
```
