1. Add a function to load recent jobs:
```javascript
async function loadRecentJobs() {
    const tableBody = document.getElementById('recent-jobs-table-body');

    try {
        const response = await fetch('/api/jobs/recent');
        const data = await response.json();

        if (data.jobs && data.jobs.length > 0) {
            tableBody.innerHTML = data.jobs.map(job => `
                <tr class="${job.status === 'running' ? 'bg-blue-50 dark:bg-blue-900/20' : ''}">
                    <td class="px-6 py-4 text-gray-800 dark:text-gray-100">${escapeHtml(job.list_name || 'Unknown')}</td>
                    <td class="px-6 py-4 text-gray-600 dark:text-gray-300">${job.target_service || '—'}</td>
                    <td class="px-6 py-4 text-gray-600 dark:text-gray-300">${formatDate(job.started_at)}</td>
                    <td class="px-6 py-4 text-gray-600 dark:text-gray-300">${formatJobSummary(job)}</td>
                </tr>
            `).join('');
        } else {
            tableBody.innerHTML = `
                <tr>
                    <td colspan="4" class="px-6 py-8 text-center text-gray-500 dark:text-gray-400">
                        No jobs have been executed yet
                    </td>
                </tr>
            `;
        }

        // Check if any jobs are running for polling
        const hasRunningJobs = data.jobs && data.jobs.some(j => j.status === 'running');
        if (hasRunningJobs && !jobsPollingInterval) {
            startJobsPolling();
        } else if (!hasRunningJobs && jobsPollingInterval) {
            stopJobsPolling();
        }

    } catch (error) {
        console.error('Error loading recent jobs:', error);
        tableBody.innerHTML = `
            <tr>
                <td colspan="4" class="px-6 py-8 text-center text-red-500 dark:text-red-400">
                    Error loading job history
                </td>
            </tr>
        `;
    }
}

function formatJobSummary(job) {
    if (job.status === 'running') {
        return '<span class="inline-flex items-center"><svg class="w-4 h-4 mr-1 animate-spin text-blue-500" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"></path></svg>Running...</span>';
    }
    if (job.status === 'failed') {
        const msg = job.error_message || 'Failed';
        return `<span class="text-red-600 dark:text-red-400">${escapeHtml(msg.substring(0, 50))}${msg.length > 50 ? '...' : ''}</span>`;
    }
    // Completed
    return `<span class="text-green-600 dark:text-green-400">${job.items_added || 0} added</span>, <span class="text-gray-500">${job.items_skipped || 0} skipped</span>`;
}

function formatDate(isoString) {
    if (!isoString) return '—';
    const date = new Date(isoString);
    // Use relative time for recent, absolute for older
    const now = new Date();
    const diffMs = now - date;
    const diffMins = Math.floor(diffMs / 60000);

    if (diffMins < 1) return 'Just now';
    if (diffMins < 60) return `${diffMins}m ago`;
    if (diffMins < 1440) return `${Math.floor(diffMins / 60)}h ago`;
    return date.toLocaleDateString();
}

function escapeHtml(text) {
    if (!text) return '';
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}

// Polling for running jobs
let jobsPollingInterval = null;

function startJobsPolling() {
    if (jobsPollingInterval) return;
    jobsPollingInterval = setInterval(loadRecentJobs, 3000);
}

function stopJobsPolling() {
    if (jobsPollingInterval) {
        clearInterval(jobsPollingInterval);
        jobsPollingInterval = null;
    }
}
```
