```javascript
/**
 * Jobs page functionality
 * Handles job listing, filtering, pagination, expandable rows, and actions
 */

// State
let currentPage = 1;
let totalPages = 1;
let currentListFilter = '';
let currentStatusFilter = '';
const perPage = 25;
let runningJobs = new Set();
let pollingInterval = null;

// DOM Elements
const tableBody = document.getElementById('jobs-table-body');
const filterList = document.getElementById('filter-list');
const filterStatus = document.getElementById('filter-status');
const applyFiltersBtn = document.getElementById('apply-filters-btn');
const clearAllBtn = document.getElementById('clear-all-btn');
const prevPageBtn = document.getElementById('prev-page-btn');
const nextPageBtn = document.getElementById('next-page-btn');
const pageInfo = document.getElementById('page-info');
const showingStart = document.getElementById('showing-start');
const showingEnd = document.getElementById('showing-end');
const totalJobsEl = document.getElementById('total-jobs');

// Initialize
document.addEventListener('DOMContentLoaded', () => {
    loadLists();
    loadJobs();
    startPolling();

    // Event listeners
    applyFiltersBtn.addEventListener('click', applyFilters);
    clearAllBtn.addEventListener('click', clearAllJobs);
    prevPageBtn.addEventListener('click', () => changePage(-1));
    nextPageBtn.addEventListener('click', () => changePage(1));
});

// Load lists for filter dropdown
async function loadLists() {
    try {
        const response = await fetch('/api/lists');
        if (response.ok) {
            const data = await response.json();
            data.lists.forEach(list => {
                const option = document.createElement('option');
                option.value = list.id;
                option.textContent = list.name;
                filterList.appendChild(option);
            });
        }
    } catch (error) {
        console.error('Error loading lists:', error);
    }
}

// Load jobs with current filters
async function loadJobs() {
    const params = new URLSearchParams({
        page: currentPage,
        per_page: perPage
    });

    if (currentListFilter) params.append('list_id', currentListFilter);
    if (currentStatusFilter) params.append('status', currentStatusFilter);

    try {
        const response = await fetch(`/api/jobs?${params}`);
        const data = await response.json();

        renderJobs(data.jobs);
        updatePagination(data);

        // Track running jobs for polling
        runningJobs = new Set(data.jobs.filter(j => j.status === 'running').map(j => j.id));
    } catch (error) {
        console.error('Error loading jobs:', error);
        showToast('Error loading jobs', 'error');
    }
}

// Render jobs table
function renderJobs(jobs) {
    if (jobs.length === 0) {
        tableBody.innerHTML = `
            <tr>
                <td colspan="7" class="px-6 py-8 text-center text-gray-500 dark:text-gray-400">
                    No jobs found
                </td>
            </tr>
        `;
        return;
    }

    tableBody.innerHTML = jobs.map(job => `
        <tr class="job-row hover:bg-gray-50 dark:hover:bg-gray-700/50" data-job-id="${job.id}">
            <td class="px-3 py-4">
                <button class="expand-btn text-gray-500 hover:text-gray-700 dark:hover:text-gray-300" onclick="toggleExpand(${job.id})">
                    <svg class="w-5 h-5 transform transition-transform" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"></path>
                    </svg>
                </button>
            </td>
            <td class="px-6 py-4 font-medium text-gray-900 dark:text-gray-100">${escapeHtml(job.list_name || 'Unknown')}</td>
            <td class="px-6 py-4">${renderStatus(job.status)}</td>
            <td class="px-6 py-4 text-gray-600 dark:text-gray-300">${formatDate(job.started_at)}</td>
            <td class="px-6 py-4 text-gray-600 dark:text-gray-300">${formatDuration(job.duration)}</td>
            <td class="px-6 py-4 text-gray-600 dark:text-gray-300">${renderResults(job)}</td>
            <td class="px-6 py-4">${renderActions(job)}</td>
        </tr>
        <tr class="job-details hidden" id="details-${job.id}">
            <td colspan="7" class="px-6 py-4 bg-gray-50 dark:bg-gray-700/50">
                <div class="loading-details text-center text-gray-500">Loading details...</div>
            </td>
        </tr>
    `).join('');
}

// Render status badge
function renderStatus(status) {
    const styles = {
        running: 'bg-blue-100 text-blue-800 dark:bg-blue-900 dark:text-blue-200',
        completed: 'bg-green-100 text-green-800 dark:bg-green-900 dark:text-green-200',
        failed: 'bg-red-100 text-red-800 dark:bg-red-900 dark:text-red-200'
    };
    const style = styles[status] || 'bg-gray-100 text-gray-800';
    const icon = status === 'running' ? '<svg class="w-3 h-3 mr-1 animate-spin" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4z"></path></svg>' : '';
    return `<span class="inline-flex items-center px-2.5 py-0.5 rounded-full text-xs font-medium ${style}">${icon}${status}</span>`;
}

// Render results summary
function renderResults(job) {
    if (job.status === 'running') return '—';
    if (job.status === 'failed' && !job.items_found) return `<span class="text-red-600 dark:text-red-400">${escapeHtml(job.error_message || 'Error')}</span>`;

    return `
        <span class="text-green-600 dark:text-green-400">${job.items_added || 0} added</span> /
        <span class="text-gray-500">${job.items_skipped || 0} skipped</span> /
        <span class="text-red-600 dark:text-red-400">${job.items_failed || 0} failed</span>
    `;
}

// Render action buttons
function renderActions(job) {
    let actions = [];

    if (job.status === 'failed') {
        actions.push(`
            <button onclick="rerunJob(${job.id})" class="text-primary hover:text-primary-dark" title="Rerun job">
                <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M4 4v5h.582m15.356 2A8.001 8.001 0 004.582 9m0 0H9m11 11v-5h-.581m0 0a8.003 8.003 0 01-15.357-2m15.357 2H15"></path>
                </svg>
            </button>
        `);
    }

    return actions.join('') || '—';
}

// Toggle expand row
async function toggleExpand(jobId) {
    const detailsRow = document.getElementById(`details-${jobId}`);
    const expandBtn = document.querySelector(`[data-job-id="${jobId}"] .expand-btn svg`);

    if (detailsRow.classList.contains('hidden')) {
        // Expand
        detailsRow.classList.remove('hidden');
        expandBtn.classList.add('rotate-90');

        // Load details
        try {
            const response = await fetch(`/api/jobs/${jobId}`);
            const job = await response.json();
            renderJobDetails(jobId, job);
        } catch (error) {
            console.error('Error loading job details:', error);
            detailsRow.querySelector('td').innerHTML = '<div class="text-red-500">Error loading details</div>';
        }
    } else {
        // Collapse
        detailsRow.classList.add('hidden');
        expandBtn.classList.remove('rotate-90');
    }
}

// Render job details
function renderJobDetails(jobId, job) {
    const detailsRow = document.getElementById(`details-${jobId}`);
    const items = job.items || [];

    let html = '<div class="space-y-4">';

    // Error details (if failed)
    if (job.status === 'failed' && job.error_details) {
        html += `
            <div class="bg-red-50 dark:bg-red-900/20 border border-red-200 dark:border-red-800 rounded p-3">
                <div class="text-sm font-medium text-red-800 dark:text-red-200 mb-1">Error Details</div>
                <pre class="text-xs text-red-700 dark:text-red-300 overflow-x-auto whitespace-pre-wrap">${escapeHtml(job.error_details)}</pre>
            </div>
        `;
    }

    // Items table
    if (items.length > 0) {
        html += `
            <div class="text-sm font-medium text-gray-700 dark:text-gray-300 mb-2">Items (${items.length})</div>
            <div class="max-h-64 overflow-y-auto">
                <table class="min-w-full text-xs">
                    <thead class="bg-gray-100 dark:bg-gray-600 sticky top-0">
                        <tr>
                            <th class="px-3 py-2 text-left">Title</th>
                            <th class="px-3 py-2 text-left">TMDB ID</th>
                            <th class="px-3 py-2 text-left">Status</th>
                            <th class="px-3 py-2 text-left">Message</th>
                        </tr>
                    </thead>
                    <tbody class="divide-y divide-gray-200 dark:divide-gray-600">
                        ${items.map(item => `
                            <tr>
                                <td class="px-3 py-2 text-gray-900 dark:text-gray-100">${escapeHtml(item.title || 'Unknown')}</td>
                                <td class="px-3 py-2 text-gray-600 dark:text-gray-300">${item.tmdb_id || '—'}</td>
                                <td class="px-3 py-2">${renderItemStatus(item.status)}</td>
                                <td class="px-3 py-2 text-gray-600 dark:text-gray-300">${escapeHtml(item.message || '—')}</td>
                            </tr>
                        `).join('')}
                    </tbody>
                </table>
            </div>
        `;
    } else {
        html += '<div class="text-gray-500 dark:text-gray-400">No items recorded</div>';
    }

    html += '</div>';
    detailsRow.querySelector('td').innerHTML = html;
}

// Render item status badge
function renderItemStatus(status) {
    const styles = {
        added: 'bg-green-100 text-green-800 dark:bg-green-900 dark:text-green-200',
        skipped: 'bg-gray-100 text-gray-800 dark:bg-gray-600 dark:text-gray-200',
        failed: 'bg-red-100 text-red-800 dark:bg-red-900 dark:text-red-200'
    };
    const style = styles[status] || 'bg-gray-100 text-gray-800';
    return `<span class="px-2 py-0.5 rounded text-xs font-medium ${style}">${status}</span>`;
}

// Rerun failed job
async function rerunJob(jobId) {
    try {
        const response = await fetch(`/api/jobs/${jobId}/rerun`, { method: 'POST' });
        const data = await response.json();

        if (data.success) {
            showToast('Job restarted', 'success');
            loadJobs(); // Refresh the list
        } else {
            showToast(data.error || 'Failed to rerun job', 'error');
        }
    } catch (error) {
        console.error('Error rerunning job:', error);
        showToast('Error rerunning job', 'error');
    }
}

// Apply filters
function applyFilters() {
    currentListFilter = filterList.value;
    currentStatusFilter = filterStatus.value;
    currentPage = 1;
    loadJobs();
}

// Clear all jobs
async function clearAllJobs() {
    if (!confirm('Are you sure you want to clear all job history? Running jobs will not be affected.')) {
        return;
    }

    try {
        const response = await fetch('/api/jobs/clear', { method: 'POST' });
        const data = await response.json();

        if (data.success) {
            showToast(`Cleared ${data.deleted_count} jobs`, 'success');
            loadJobs();
        } else {
            showToast('Failed to clear jobs', 'error');
        }
    } catch (error) {
        console.error('Error clearing jobs:', error);
        showToast('Error clearing jobs', 'error');
    }
}

// Pagination
function updatePagination(data) {
    totalPages = data.pages || 1;
    const total = data.total || 0;
    const start = total > 0 ? (currentPage - 1) * perPage + 1 : 0;
    const end = Math.min(currentPage * perPage, total);

    showingStart.textContent = start;
    showingEnd.textContent = end;
    totalJobsEl.textContent = total;
    pageInfo.textContent = `Page ${currentPage} of ${totalPages}`;

    prevPageBtn.disabled = currentPage <= 1;
    nextPageBtn.disabled = currentPage >= totalPages;
}

function changePage(delta) {
    const newPage = currentPage + delta;
    if (newPage >= 1 && newPage <= totalPages) {
        currentPage = newPage;
        loadJobs();
    }
}

// Polling for running jobs
function startPolling() {
    pollingInterval = setInterval(() => {
        if (runningJobs.size > 0) {
            loadJobs();
        }
    }, 3000); // Poll every 3 seconds when jobs are running
}

// Utilities
function formatDate(isoString) {
    if (!isoString) return '—';
    const date = new Date(isoString);
    return date.toLocaleString();
}

function formatDuration(seconds) {
    if (!seconds && seconds !== 0) return '—';
    if (seconds < 60) return `${seconds}s`;
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins}m ${secs}s`;
}

function escapeHtml(text) {
    if (!text) return '';
    const div = document.createElement('div');
    div.textContent = text;
    return div.innerHTML;
}

function showToast(message, type = 'info') {
    const container = document.getElementById('toast-container');
    const colors = {
        success: 'bg-green-500',
        error: 'bg-red-500',
        info: 'bg-blue-500'
    };

    const toast = document.createElement('div');
    toast.className = `${colors[type] || colors.info} text-white px-4 py-2 rounded-lg shadow-lg transition-opacity duration-300`;
    toast.textContent = message;

    container.appendChild(toast);

    setTimeout(() => {
        toast.classList.add('opacity-0');
        setTimeout(() => toast.remove(), 300);
    }, 3000);
}
```
