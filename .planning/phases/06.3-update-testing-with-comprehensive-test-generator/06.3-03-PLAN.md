class TestRegionAwareCacheKeys:
    """Tests for region-aware cache key generation (Phase 6.2)."""

    @patch('listarr.services.tmdb_cache.get_top_rated_movies')
    @patch('listarr.services.tmdb_cache._get_tmdb_region')
    def test_different_regions_use_different_cache_keys(self, mock_region, mock_fetch, app):
        """Different regions result in separate cache entries."""
        with app.app_context():
            # First call with US region
            mock_region.return_value = 'US'
            mock_fetch.return_value = [{'id': 1, 'title': 'US Movie'}]
            result_us = get_top_rated_movies_cached("test_key", page=1)

            # Clear the mock to track new calls
            mock_fetch.reset_mock()

            # Second call with GB region should fetch again (different cache key)
            mock_region.return_value = 'GB'
            mock_fetch.return_value = [{'id': 2, 'title': 'GB Movie'}]
            result_gb = get_top_rated_movies_cached("test_key", page=1)

        # Both regions should have fetched (cache keys are different)
        assert mock_fetch.call_count == 1  # Second fetch happened
        assert result_us != result_gb

    @patch('listarr.services.tmdb_cache.get_top_rated_movies')
    @patch('listarr.services.tmdb_cache._get_tmdb_region')
    def test_same_region_uses_cached_result(self, mock_region, mock_fetch, app):
        """Same region reuses cached result."""
        mock_region.return_value = 'FR'
        mock_fetch.return_value = [{'id': 1, 'title': 'French Movie'}]

        with app.app_context():
            result1 = get_top_rated_movies_cached("test_key", page=1)
            result2 = get_top_rated_movies_cached("test_key", page=1)

        assert result1 == result2
        assert mock_fetch.call_count == 1

    @patch('listarr.services.tmdb_cache.get_top_rated_movies')
    @patch('listarr.services.tmdb_cache._get_tmdb_region')
    def test_no_region_uses_worldwide_cache_key(self, mock_region, mock_fetch, app):
        """No region configured uses worldwide cache key."""
        mock_region.return_value = None
        mock_fetch.return_value = [{'id': 1, 'title': 'Worldwide Movie'}]

        with app.app_context():
            result1 = get_top_rated_movies_cached("test_key", page=1)
            result2 = get_top_rated_movies_cached("test_key", page=1)

        assert result1 == result2
        assert mock_fetch.call_count == 1
